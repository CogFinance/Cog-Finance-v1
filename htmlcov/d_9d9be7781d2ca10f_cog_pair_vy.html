24@internal 

25def to_base_round_up(total: Rebase, elastic: uint256) -> uint256: 

26 """ 

27 @param total - The Rebase value which should be used to dervie the relative base value 

28 @param elastic - The elastic value to convert to a relative base value 

29 @param round_up - Self explanatory 

30 """ 

31 if total.elastic == 0: 

32 # If elastic is 0, then 0/n = 0 ∀ n ∈ R 

33 return elastic 

34 else: 

35 # Base is equal to elastic * (total.base / total.elastic), essentially a ratio 

36 base: uint256 = (elastic * convert(total.base, uint256)) / convert( 

37 total.elastic, uint256 

38 ) 

39  

40 # mamushi: Exhibit 1 

41 if ( 

42 ( 

43 (base * convert(total.elastic, uint256)) 

44 / convert(total.base, uint256) 

45 ) 

46 < elastic 

47 ): 

48 base = base + 1 

49  

50 return base

54@internal 

55def to_elastic(total: Rebase, base: uint256, round_up: bool) -> uint256: 

56 """ 

57 @param total - The Rebase which should be used to derive the relative elastic value 

58 @param base - The base value to convert to a relative elastic value 

59 @param round_up - Self explanatory 

60 """ 

61 if total.base == 0: 

62 # If base is 0, then Rebase would be n/n, so elastic = base 

63 return base 

64 else: 

65 # Elastic is equal to base * (total.elastic / total.base), essentially a ratio 

66 elastic: uint256 = (base * convert(total.elastic, uint256)) / convert( 

67 total.base, uint256 

68 ) 

69  

70 # mamushi: Exhibit 2 

71 if round_up and ( 

72 ( 

73 (elastic * convert(total.base, uint256)) 

74 / convert(total.elastic, uint256) 

75 ) 

76 < base 

77 ): 

78 elastic = elastic + 1 

79  

80 return elastic

84def add_round_up(_total: Rebase, elastic: uint256) -> (Rebase, uint256): 

85 """ 

86 @notice Add `elastic` to `total` and doubles `total.base`

88 @param _total - The current total

120# Math Helper For Precision # 

121# ///////////////////////////////////////////////////// # 

122# Ty snekmate again 

123@pure 

124@internal 

125def mul_div( 

126 x: uint256, y: uint256, denominator: uint256, roundup: bool 

127) -> uint256: 

128 """ 

129 @dev Calculates "(x * y) / denominator" in 512-bit precision, 

130 following the selected rounding direction. 

131 @notice The implementation is inspired by Remco Bloemen's 

132 implementation under the MIT license here: 

133 https://xn--2-umb.com/21/muldiv. 

134 Furthermore, the rounding direction design pattern is 

135 inspired by OpenZeppelin's implementation here: 

136 https://github.com/OpenZeppelin/openzeppelin-contracts/blob/master/contracts/utils/math/Math.sol. 

137 @param x The 32-byte multiplicand. 

138 @param y The 32-byte multiplier. 

139 @param denominator The 32-byte divisor. 

140 @param roundup The Boolean variable that specifies whether 

141 to round up or not. The default `False` is round down. 

142 @return uint256 The 32-byte calculation result. 

143 """ 

144 # Handle division by zero. 

145 assert denominator != empty(uint256), "Math: mul_div division by zero" 

146  

147 # 512-bit multiplication "[prod1 prod0] = x * y". 

148 # Compute the product "mod 2**256" and "mod 2**256 - 1". 

149 # Then use the Chinese Remainder theorem to reconstruct 

150 # the 512-bit result. The result is stored in two 256-bit 

151 # variables, where: "product = prod1 * 2**256 + prod0". 

152 mm: uint256 = uint256_mulmod(x, y, max_value(uint256)) 

153 # The least significant 256 bits of the product. 

154 prod0: uint256 = unsafe_mul(x, y) 

155 # The most significant 256 bits of the product. 

156 prod1: uint256 = empty(uint256) 

157  

158 if mm < prod0: 

159 prod1 = unsafe_sub(unsafe_sub(mm, prod0), 1) 

160 else: 

161 prod1 = unsafe_sub(mm, prod0) 

162  

163 if prod1 == empty(uint256): 

164 if roundup and uint256_mulmod(x, y, denominator) != empty(uint256): 

165 # Calculate "ceil((x * y) / denominator)". The following 

166 # line cannot overflow because we have the previous check 

167 # "(x * y) % denominator != 0", which accordingly rules out 

168 # the possibility of "x * y = 2**256 - 1" and `denominator == 1`. 

169 return unsafe_add(unsafe_div(prod0, denominator), 1) 

170 else: 

171 return unsafe_div(prod0, denominator) 

172  

173 # Ensure that the result is less than 2**256. Also, 

174 # prevents that `denominator == 0`. 

175 assert denominator > prod1, "Math: mul_div overflow" 

176  

177 ####################### 

178 # 512 by 256 Division # 

179 ####################### 

180  

181 # Make division exact by subtracting the remainder 

182 # from "[prod1 prod0]". First, compute remainder using 

183 # the `uint256_mulmod` operation. 

184 remainder: uint256 = uint256_mulmod(x, y, denominator) 

185  

186 # Second, subtract the 256-bit number from the 512-bit 

187 # number. 

188 if remainder > prod0: 

189 prod1 = unsafe_sub(prod1, 1) 

190 prod0 = unsafe_sub(prod0, remainder) 

191  

192 # Factor powers of two out of the denominator and calculate 

193 # the largest power of two divisor of denominator. Always `>= 1`, 

194 # unless the denominator is zero (which is prevented above), 

195 # in which case `twos` is zero. For more details, please refer to: 

196 # https://cs.stackexchange.com/q/138556. 

197  

198 # The following line does not overflow because the denominator 

199 # cannot be zero at this stage of the function. 

200 twos: uint256 = denominator & (unsafe_add(~denominator, 1)) 

201 # Divide denominator by `twos`. 

202 denominator_div: uint256 = unsafe_div(denominator, twos) 

203 # Divide "[prod1 prod0]" by `twos`. 

204 prod0 = unsafe_div(prod0, twos) 

205 # Flip `twos` such that it is "2**256 / twos". If `twos` is zero, 

206 # it becomes one. 

207 twos = unsafe_add(unsafe_div(unsafe_sub(empty(uint256), twos), twos), 1) 

208  

209 # Shift bits from `prod1` to `prod0`. 

210 prod0 |= unsafe_mul(prod1, twos) 

211  

212 # Invert the denominator "mod 2**256". Since the denominator is 

213 # now an odd number, it has an inverse modulo 2**256, so we have: 

214 # "denominator * inverse = 1 mod 2**256". Calculate the inverse by 

215 # starting with a seed that is correct for four bits. That is, 

216 # "denominator * inverse = 1 mod 2**4". 

217 inverse: uint256 = unsafe_mul(3, denominator_div) ^ 2 

218  

219 # Use Newton-Raphson iteration to improve accuracy. Thanks to Hensel's 

220 # lifting lemma, this also works in modular arithmetic by doubling the 

221 # correct bits in each step. 

222 inverse = unsafe_mul( 

223 inverse, unsafe_sub(2, unsafe_mul(denominator_div, inverse)) 

224 ) # Inverse "mod 2**8". 

225 inverse = unsafe_mul( 

226 inverse, unsafe_sub(2, unsafe_mul(denominator_div, inverse)) 

227 ) # Inverse "mod 2**16". 

228 inverse = unsafe_mul( 

229 inverse, unsafe_sub(2, unsafe_mul(denominator_div, inverse)) 

230 ) # Inverse "mod 2**32". 

231 inverse = unsafe_mul( 

232 inverse, unsafe_sub(2, unsafe_mul(denominator_div, inverse)) 

233 ) # Inverse "mod 2**64". 

234 inverse = unsafe_mul( 

235 inverse, unsafe_sub(2, unsafe_mul(denominator_div, inverse)) 

236 ) # Inverse "mod 2**128". 

237 inverse = unsafe_mul( 

238 inverse, unsafe_sub(2, unsafe_mul(denominator_div, inverse)) 

239 ) # Inverse "mod 2**256". 

240  

241 # Since the division is now exact, we can divide by multiplying 

242 # with the modular inverse of the denominator. This returns the 

243 # correct result modulo 2**256. Since the preconditions guarantee 

244 # that the result is less than 2**256, this is the final result. 

245 # We do not need to calculate the high bits of the result and 

246 # `prod1` is no longer necessary. 

247 result: uint256 = unsafe_mul(prod0, inverse) 

248  

249 if roundup and uint256_mulmod(x, y, denominator) != empty(uint256): 

250 # Calculate "ceil((x * y) / denominator)". The following 

251 # line uses intentionally checked arithmetic to prevent 

252 # a theoretically possible overflow. 

253 result += 1 

254  

255 return result

507def _convertToAssets(shareAmount: uint256) -> uint256: 

508 _total_asset: Rebase = self.total_asset 

509 if _total_asset.base == 0:

530def _convertToShares(assetAmount: uint256) -> uint256:

710def efficient_accrue(): 

711 _accrue_info: AccrueInfo = self.accrue_info 

712 elapsed_time: uint256 = block.timestamp - convert( 

713 _accrue_info.last_accrued, uint256 

714 ) 

715 if elapsed_time == 0: 

716 # Prevents re-executing this logic if multiple actions are taken in the same block 

717 return 

718  

719 self._accrue(_accrue_info, elapsed_time)

721@internal 

722def _accrue(accrue_info: AccrueInfo, elapsed_time: uint256): 

723 _accrue_info: AccrueInfo = accrue_info 

724 _accrue_info.last_accrued = convert(block.timestamp, uint64) 

725  

726 _total_borrow: Rebase = self.total_borrow 

727 if _total_borrow.base == 0: 

728 # If there are no outstanding borrows, there is no need to accrue interest, and interest 

729 # rate should be moved to minimum to encourage borrowing 

730 if _accrue_info.interest_per_second != STARTING_INTEREST_PER_SECOND: 

731 _accrue_info.interest_per_second = STARTING_INTEREST_PER_SECOND 

732 self.accrue_info = _accrue_info 

733 return 

734  

735 interest_accrued: uint256 = 0 

736 fee_fraction: uint256 = 0 

737 _total_asset: Rebase = self.total_asset 

738  

739 # Accrue interest 

740 interest_accrued = ( 

741 convert(_total_borrow.elastic, uint256) 

742 * convert(_accrue_info.interest_per_second, uint256) 

743 * elapsed_time 

744 / 1000000000000000000 

745 ) # 1e18, or the divisor for interest per second 

746  

747 _total_borrow.elastic = _total_borrow.elastic + convert( 

748 interest_accrued, uint128 

749 ) 

750  

751 full_asset_amount: uint256 = convert( 

752 _total_asset.elastic, uint256 

753 ) + convert(_total_borrow.elastic, uint256) 

754  

755 # Calculate fees 

756 fee_amount: uint256 = ( 

757 interest_accrued * self.protocol_fee / PROTOCOL_FEE_DIVISOR 

758 ) # % of interest paid goes to fee 

759  

760 fee_fraction = ( 

761 fee_amount * convert(_total_asset.base, uint256) / full_asset_amount 

762 ) # Update total fees earned 

763 _accrue_info.fees_earned_fraction = ( 

764 _accrue_info.fees_earned_fraction + convert(fee_fraction, uint128) 

765 ) 

766  

767 # Fees should be considered in total assets 

768 self.total_asset.base = _total_asset.base + convert(fee_fraction, uint128) 

769  

770 # Write new total borrow state to storage 

771 self.total_borrow = _total_borrow 

772  

773 # Update interest rate 

774 utilization: uint256 = ( 

775 convert(_total_borrow.elastic, uint256) 

776 * UTILIZATION_PRECISION 

777 / full_asset_amount 

778 ) 

779  

780 if utilization < MINIMUM_TARGET_UTILIZATION: 

781 under_factor: uint256 = ( 

782 (MINIMUM_TARGET_UTILIZATION - utilization) 

783 * FACTOR_PRECISION 

784 / MINIMUM_TARGET_UTILIZATION 

785 ) 

786 scale: uint256 = INTEREST_ELASTICITY + ( 

787 under_factor * under_factor * elapsed_time 

788 ) 

789 new_interest_per_second: uint64 = convert( 

790 convert(_accrue_info.interest_per_second, uint256) 

791 * INTEREST_ELASTICITY 

792 / scale, 

793 uint64, 

794 ) 

795 _accrue_info.interest_per_second = new_interest_per_second 

796  

797 if _accrue_info.interest_per_second < MINIMUM_INTEREST_PER_SECOND: 

798 _accrue_info.interest_per_second = ( 

799 MINIMUM_INTEREST_PER_SECOND # 0.25% APR minimum 

800 ) 

801 elif utilization > MAXIMUM_TARGET_UTILIZATION: 

802 over_factor: uint256 = ( 

803 (utilization - MAXIMUM_TARGET_UTILIZATION) 

804 * FACTOR_PRECISION 

805 / MAXIMUM_TARGET_UTILIZATION 

806 ) 

807 scale: uint256 = INTEREST_ELASTICITY + ( 

808 over_factor * over_factor * elapsed_time 

809 ) 

810 new_interest_per_second: uint64 = convert( 

811 convert(_accrue_info.interest_per_second, uint256) 

812 * scale 

813 / INTEREST_ELASTICITY, 

814 uint64, 

815 ) 

816  

817 if new_interest_per_second > MAXIMUM_INTEREST_PER_SECOND: 

818 _accrue_info.interest_per_second = ( 

819 MAXIMUM_INTEREST_PER_SECOND # 100% APR maximum 

820 ) 

821  

822 _accrue_info.interest_per_second = new_interest_per_second 

823  

824 dt: uint64 = ( 

825 convert(block.timestamp, uint64) - self.surge_info.last_elapsed_time 

826 ) 

827 if dt > 86400: 

828 # if interest rate is increasing 

829 if ( 

830 _accrue_info.interest_per_second 

831 > self.surge_info.last_interest_per_second 

832 ): 

833 # If daily change in interest rate is greater than Surge threshold, trigger surge breaker 

834 dr: uint64 = ( 

835 _accrue_info.interest_per_second 

836 - self.surge_info.last_interest_per_second 

837 ) 

838 if dr > PROTOCOL_SURGE_THRESHOLD: 

839 self.surge_info.last_elapsed_time = convert( 

840 block.timestamp, uint64 

841 ) 

842 self.surge_info.last_interest_per_second = ( 

843 _accrue_info.interest_per_second 

844 ) 

845 # PoL Should accrue here, instead of to lenders, to discourage pid attacks as described in https://gauntlet.network/reports/pid 

846 self.protocol_fee = PROTOCOL_FEE_DIVISOR # 100% Protocol Fee 

847 else: 

848 # Reset protocol fee elsewise 

849 self.protocol_fee = self.DEFAULT_PROTOCOL_FEE # 10% Protocol Fee 

850 self.accrue_info = _accrue_info
